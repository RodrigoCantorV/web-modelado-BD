Para dar solución a target="_top" este problema podríamos hacer que el campo fecha se comporte como otra clave principal y de esta forma la integridad de los datos se mantendría. 						DROP DATABASE IF EXISTS OBRAS_TEATRALES;
						CREATE DATABASE IF NOT EXISTS OBRAS_TEATRALES;
						USE OBRAS_TEATRALES;
						
						CREATE TABLE AUTOR(
						id INT AUTO_INCREMENT PRIMARY KEY,
						nombres VARCHAR(40) NOT NULL,
						apellidos VARCHAR(50) NOT NULL,
						fecha_nacimiento DATE NULL,
						pais VARCHAR(20)
						);
	
						CREATE TABLE IF NOT EXISTS OBRA(
						id INT AUTO_INCREMENT NOT NULL,
						nombre VARCHAR(40) NOT NULL,
						genero ENUM('COMEDIA', 'DRAMA', 'MUSICAL') NOT NULL,
						sinopsis TEXT,
						duracion TIME DEFAULT '01:00:00',
						id_autor INT NOT NULL,
						PRIMARY KEY (id),
						FOREIGN KEY (id_autor) REFERENCES AUTOR (id) ON UPDATE CASCADE
						);
				
						CREATE TABLE IF NOT EXISTS FUNCION(
						id INT AUTO_INCREMENT PRIMARY KEY NOT NULL,
						fecha DATE NOT NULL,
						hora TIME NOT NULL,
						lugar VARCHAR(50) NOT NULL,
						precio FLOAT NOT NULL,
						id_obra INT NOT NULL,
						FOREIGN KEY (id_obra) REFERENCES OBRA (id)
						);
				
						CREATE TABLE IF NOT EXISTS PERSONAJE(
						id_obra INT NOT NULL,
						nombre_personaje VARCHAR(20) NOT NULL,
						genero ENUM('M', 'F', 'O') NOT NULL,
						PRIMARY KEY (id_obra, nombre_personaje),
						CONSTRAINT FK_ID_OBRA_PERS FOREIGN KEY (id_obra) REFERENCES OBRA (id)
						);
				
						CREATE TABLE IF NOT EXISTS ACTOR(
						id INT AUTO_INCREMENT PRIMARY KEY NOT NULL,
						nombre VARCHAR(40) NOT NULL,
						apellidos VARCHAR(50) NOT NULL,
						fecha_nacimiento DATE NULL,
						genero ENUM('M', 'F') DEFAULT 'F'
						);
				
						CREATE TABLE IF NOT EXISTS REPRESENTACION(
						id_actor INT NOT NULL,
						id_obra INT NOT NULL,
						nombre_personaje VARCHAR(20) NOT NULL,
						fecha_actualizacion TIMESTAMP DEFAULT NOW(),
						PRIMARY KEY(id_actor, id_obra, nombre_personaje),
						FOREIGN KEY (id_actor) REFERENCES ACTOR (id) ON UPDATE CASCADE,
						FOREIGN KEY (id_obra, nombre_personaje) REFERENCES PERSONAJE (id_obra, nombre_personaje)
						);
